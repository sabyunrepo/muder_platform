```go
type ReadingLine struct {
    Index        int    `json:"index"`
    Text         string `json:"text"`
    Speaker      string `json:"speaker,omitempty"`
    VoiceMediaID string `json:"voiceMediaId,omitempty"`
    AdvanceBy    string `json:"advanceBy"` // "voice" | "gm" | "role:<roleId>"
}

func (m *ReadingModule) HandleAdvance(sess *Session, userID string) error {
    line := m.currentLine()
    if line == nil {
        return apperror.NotFound(apperror.ErrReadingLineOutOfRange, "no current line")
    }
    switch {
    case line.AdvanceBy == "voice":
        return apperror.Forbidden(apperror.ErrReadingAdvanceForbidden, "voice mode")
    case line.AdvanceBy == "gm":
        if userID != sess.HostID {
            return apperror.Forbidden(apperror.ErrReadingAdvanceForbidden, "gm only")
        }
    case strings.HasPrefix(line.AdvanceBy, "role:"):
        roleID := strings.TrimPrefix(line.AdvanceBy, "role:")
        if sess.PlayerRoles[userID] != roleID {
            return apperror.Forbidden(apperror.ErrReadingAdvanceForbidden, "role mismatch").
                WithParams(map[string]any{"requiredRole": roleID})
        }
    default:
        return apperror.Validation(apperror.ErrReadingInvalidAdvanceBy, line.AdvanceBy)
    }
    m.advance()
    return nil
}
```

```ts
// audioGraph.ts
export type AudioChannel = 'master' | 'bgm' | 'voice' | 'sfx'

export interface AudioGraph {
  ctx: AudioContext
  masterGain: GainNode
  channelGains: Record<Exclude<AudioChannel, 'master'>, GainNode>
  getGainNode(channel: AudioChannel): GainNode
  setChannelVolume(channel: AudioChannel, volume: number): void
  dispose(): void
}

export function createAudioGraph(ctx: AudioContext): AudioGraph {
  const master = ctx.createGain()
  master.connect(ctx.destination)
  const bgm = ctx.createGain(); bgm.connect(master)
  const voice = ctx.createGain(); voice.connect(master)
  const sfx = ctx.createGain(); sfx.connect(master)
  return { ctx, masterGain: master, channelGains: { bgm, voice, sfx }, ... }
}
```

## 주의사항 (Gotchas)

1. **기존 Phase 7.7 Step 2 코드와의 정합성**: `ReadingLine` 구조 확장 시 기존 serialize/deserialize 코드 모두 업데이트 필요. `grep -r "ReadingLine" apps/server`로 전파 확인.
2. **Web Audio AudioContext 테스트**: 브라우저 환경이 필요. 테스트용 `jsdom` 환경에서는 AudioContext 가 없으므로 mock 헬퍼 작성 필수 (`apps/web/src/test/audioContextMock.ts`).
3. **YouTube IFrame API 로드 타이밍**: `window.YT`는 비동기 로드. Orchestrator는 준비 완료까지 대기하는 promise 제공 필요.
4. **리딩 advanceBy voice + voiceMediaId 404**: 엣지 케이스 — VoiceManager가 404를 받으면 즉시 `voice_ended` 발행해 서버가 멈추지 않게 함.
5. **presence 이탈 감지**: Phase 7.5의 SocialHub 연동 코드 확인. reading 섹션이 활성일 때만 paused 발행해야 함 (항상 감시는 오버헤드).
6. **optimistic locking**: `reading_sections.version` 컬럼 활용, UPDATE 시 WHERE version=? 조건으로 충돌 감지.
7. **Tailwind CSS 경로**: 프로젝트 규칙상 Tailwind only. Seed Design 시스템 사용 금지. 기존 slate/zinc + amber 색상 유지.
8. **lucide-react 아이콘**: 스피커(Volume2/VolumeX), 플레이(Play/Pause/SkipForward), 업로드(Upload), 유튜브(Youtube) 등.
